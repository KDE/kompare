target_sources(komparedialogpages PRIVATE
    filessettings.cpp
    viewsettings.cpp
    diffpage.cpp
    filespage.cpp
    viewpage.cpp
)

)

    VERSION   ${KOMPARE_LIB_VERSION}

)
